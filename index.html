Correlation Search SPL

Déployée dans Splunk ES · Notable event auto-créé · CIM-compliant

ACTIVE risk=80 
/* Correlation Search: Suspicious MSI Execution with C2 Beacon */
/* Author: Axel Sellami · CMA PACA SOC · Date: 2026/02/26   */
/* MITRE: T1204.002 (Malicious File) · TA0011 (C2)           */

index=wineventlog sourcetype="XmlWinEventLog:Microsoft-Windows-Sysmon/Operational"
    EventCode=1 Image="*\msiexec.exe" Signed=false
    (CommandLine="*/i*" CommandLine="*.msi*")
| rename ProcessGuid as src_guid, Image as process
| join type=inner src_guid max=0 [
    search index=wineventlog sourcetype="*Sysmon*"
        EventCode=3 Initiated="true"
        NOT (DestinationIp="10.0.0.0/8" OR DestinationIp="192.168.0.0/16"
             OR DestinationIp="172.16.0.0/12")
    | rename ProcessGuid as src_guid
    | eval net_time=_time
]
| eval delta=abs(net_time - _time)
| where delta <= 120   /* 2 min timeframe */
| lookup sophos_ioc_feed DestinationIp OUTPUT threat_category, risk
| stats count, values(DestinationIp) as dest_ips, values(process) as processes
    by host, user, Computer
| eval urgency="high", risk_score=80
| where count >= 1
| collect index=notable
